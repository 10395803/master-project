	otherkeywords={self},             % Add keywords here

	emph={MyClass,__init__},          % Custom highlighting
	emphstyle=\ttb\color{deepred},    % Custom highlighting style
	stringstyle=\color{deepgreen},

	showstringspaces=false            % 
}}
